### 父组件向子组件

props

~~~js
props: {
	prop: {
        type: String,
        default: '123',
        required: true
	}
}
~~~



子组件向父组件

$emit

.sync/$emit('update:x','11')



### A <---> C组件通信（隔了2代）

`v-bind="$attrs"` `v-on="$listeners"` 作中间衔接作用

~~~vue
// Component1.vue
<template>
  <div>
    <component-2 :prop1='prop1' :prop2.sync="prop2" />
    {{`outer--${prop2}`}}
  </div>
</template>

<script>
import Component2 from "./Component2.vue";
export default {
  components: { Component2 },
  name: "OuterComponent",
  data() {
    return {
      prop1: 'propdd',
      prop2: 'feff'
    };
  },
};
</script>
~~~

~~~vue
// Component2.vue
<template>
  <div>
    <component-3 v-bind="$attrs" v-on="$listeners"/>
  </div>
</template>

<script>
import Component3 from "./Component3.vue";
export default {
  name: "Component2",
  data() {
    return {};
  },
  components:{Component3}
};
</script>
~~~

~~~vue
<template>
  <div>
    {{ prop1 }}
    {{ prop2 }}
    <button
      @click="
        () => {
          this.$emit('update:prop2', 'peop22ee');
        }
      "
    >
      changeprop2
    </button>
  </div>
</template>

<script>
export default {
  name: "Component3",
  inheritAttrs: false,
  props: {
    prop1: String,
    prop2: String,
  },
};
</script>
~~~



### provide 和 inject

大范围的props

~~~vue
// Component1.vue
<template>
  <div>
    <component-2 />
  </div>
</template>

<script>
import Component2 from "./Component2.vue";
export default {
  name: "Component1",
  components: { Component2 },
  data() {
    return {
    };
  },
  provide:{
    prop1: 'prop11from1'
  }
};
</script>
~~~

~~~vue
// Component2.vue
<template>
  <div>
    <component-3/>
  </div>
</template>

<script>
import Component3 from "./Component3.vue";
export default {
  name: "Component2",
  data() {
    return {};
  },
  components:{Component3}
};
</script>
~~~

~~~vue
// Component3.vue
<template>
  <div>
    {{ prop1 }}
  </div>
</template>

<script>
export default {
  name: "Component3",
  inject: ['prop1'],
};
</script>
~~~
